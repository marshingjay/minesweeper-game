 11 /** ------------Global variables ------------ */
 12 
 13 let grid;
 14 let cols=0;
 15 let rows=0;
 16 let mines=0;
 17 let width=30;
 18 let flags=0;
 19 
 20 /** ------------ P5 interface ------------ */
 21 /** Creates a canvas with a 2D array according to the input
 22     * @pre need a valid number of rows and columns to build the board
 23     * @post the board has been created and filled with mines, the user can now play the game!
 24     */
 25 function setup() {
 26     /** Gets the dimensions from the user */
 27     rows = floor(document.getElementById("input1").value);
 28     cols = floor(document.getElementById("input2").value);
 29     mines = floor(document.getElementById("input3").value);
 30     flags = mines;
 31 
 32     /** Boundaries for the grid */
 33     if (rows<2){
 34         rows=2;
 35     }
 36     if (cols < 2){
 37         cols = 2;
 38     }
 39     if (rows > 30){
 40         rows = 30;
 41     }
 42     if (cols > 30){
 43         cols = 30;
 44     }
 45     if(mines<=0){
 46         mines=1;
 47         flags=1;
 48     }
 49     if(mines>=rows*cols){
 50         mines=rows*cols-1;
 51         flags=mines;
 52     }
 53 
 54     /** Creates a canvas that holds the amount of cols and rows given according to the set width*/
 55     let canvas = createCanvas(cols*width +1, rows*width +1);
 56     canvas.parent('canvas-holder');
 57 
 58     /** Creates a 2D Array with the cols and rows given*/
 59     grid = create2DArray(cols, rows);
 60     for (let c = 0; c < cols; c++) {
 61       for (let r = 0; r < rows; r++) {
 62         grid[c][r] = new Box(c*width, r*width, width);
 63       }
 64     }
 65 
 66     /** Populates the grid with the amount of mines given */
 67     mine_population(mines, rows, cols, grid);
 68 
 69     /** Populates the count of each box in the grid */
 70     generate_playing_field(mines, rows, cols, grid);
 71 
 72 }
 73 
 74 /** Draws the canvas on the site by calling the show function on each box
 75     * @pre there has been a 2d array built, but it has nothing inside it
 76     * @post the array is now filled with mines or number of adjacent mines
 77     * @return none
 78     */
 79 function draw() {
 80   for (let c = 0; c < cols; c++) {
 81     for (let r = 0; r < rows; r++) {
 82       grid[c][r].show();
 83     }
 84   }
 85 }

 87 /** ------------ Helper Functions ------------ */
 88 
 89 /**
 90  * Creates a 2D Array according to the cols and rows given
 91  * @param   {number} cols number of columns for the 2D array
 92  * @param   {number} rows number of rows for the 2D array
 93  * @return  {array} 2D array with the the numbers of cols and rows
 94  */
 95 function create2DArray(cols,rows){
 96   let grid = new Array(cols);
 97   for (let c=0; c<cols; c++){
 98     grid[c] = new Array(rows);
 99   }
100   return grid;
101 }

104 /** Places a flags where the mouse is located when f is released
105   * @pre An array has been created and filled with box objects, some of which contain mines
106   * @post Once the f key has been released a flag will be set wherever the cursor hovered when f is pressed
107   *       (decrementing flagCount by 1) if this is repeated, the flag will be removed and flagCount will increase by 1
108   * @return none
109 */
110 function keyReleased(){
111         if (key === 'f'){
112                 let x;
113                 let y;
114                 x = floor(mouseX/width);
115                 y = floor(mouseY/width);
116                 if (grid[x][y].clicked==false){
117                         if (grid[x][y].flagged==true){
118                                 grid[x][y].flagged=false;
119                                 flags = flags + 1;
120                         }
121                         else if (flags > 0){
122                                 grid[x][y].flagged=true;
123                                 flags = flags - 1;
124                         }
125                 }
126                 document.getElementById("flagsLeft").innerHTML = ("Remaining flags: " + flags);
127 
128                 if(win(cols, rows, grid, mines)){
129                     window.alert("You won!");
130                     location = location;
131                 }
132         }
133 }

136 /**
137  * Changes the box implementation when it is clicked  the number of mines adjacent to that spot will appear. If they have flagged every
138  * mine, they win
139  *
140  * @pre The board has been created and filled with mines, the user is playing
141  * @post if the user clicks a mine, they lose. If they click anything other than a mine,
142  * @return none
143  */
144 function mouseClicked(){
145     /** Gets coordinate of the click input */
146     let x = floor(mouseX/width);
147     let y = floor(mouseY/width);
148     /** If the box does not have a flag, change the boxs implementation */
149     if (grid[x][y].flagged==false){
150       grid[x][y].clicked=true;
151       /** Generates spaces if the box is an space and not a mine*/
152       if (grid[x][y].count==0 && grid[x][y].mine==false){
153           reveal_spaces(x,y,cols,rows,grid);
154       }
155       /** Calls win function */
156       if(win(cols, rows, grid, mines)){
157           window.alert("You won!");
158           window.location.reload(true);
159       }
160       /** Calls lose function */
161       if(grid[x][y].mine == true){
162         lose();
163       }
164     }
165 }
